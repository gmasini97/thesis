    void butterfly_calculation(cuComplex* a, cuComplex* b, cuComplex w)
    {
        cuComplex aa = *a;
        cuComplex bw = cuCmulf(*b, w);
    
        *a = cuCaddf(aa, bw);
        *b = cuCsubf(aa, bw);
    }
